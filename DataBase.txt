A continuación adjunto los scrips para crear la base de datos, la secuencia que autoincrementa el id y la tabla 'notification'


CREATE DATABASE test_db
  WITH OWNER = postgres
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'en_US.UTF-8'
       LC_CTYPE = 'en_US.UTF-8'
       CONNECTION LIMIT = -1;



CREATE SEQUENCE public.notification_id_seq
  INCREMENT 1
  MINVALUE 1
  MAXVALUE 9223372036854775807
  START 1
  CACHE 1;
ALTER TABLE public.notification_id_seq
  OWNER TO postgres;



CREATE TABLE public.notification
(
  id integer NOT NULL DEFAULT nextval('notification_id_seq'::regclass),
  pre_scene_id integer NOT NULL,
  user_from character varying(32) NOT NULL,
  user_to character varying(32) NOT NULL,
  read boolean DEFAULT false,
  message integer NOT NULL,
  creation_date date NOT NULL,
  type integer NOT NULL DEFAULT 0,
  date date
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.notification
  OWNER TO postgres;
